        role="tablist"

          const tabClasses = cn(style.tab, {

            [style.first]: index === 0,
            [style.last]: index === this.props.items.length - 1,

          return (
            <li

              onClick={() => this.onClick(tab, index)}
              onKeyDown={(event) => this.onKeyDown(event, tab, index)}
              className={tabClasses}
              key={index}
              tabIndex="0"
            >

          );
        })}
      </ul>
    );
  }
